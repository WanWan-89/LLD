Documentation Base de Données

Script SQL de création des tables et des relations pour le système de gestion scolaire.

schema_ecole.sql 

class = (id_class SERIAL, list_of_class VARCHAR(50) );
parents = (id_parent SERIAL, name VARCHAR(50) , surname VARCHAR(50) , email VARCHAR(50) , phone_number VARCHAR(20) );
subject = (id_subject SERIAL, subject VARCHAR(50) );
teacher = (id_teacher SERIAL, name VARCHAR(50) , surname VARCHAR(50) , email VARCHAR(50) , phone_number VARCHAR(50) );
user_ = (id_user SERIAL, role VARCHAR(50) , #id_teacher);
messages = (id_message SERIAL, content VARCHAR(1000) , #id_user);
audit = (id_audit_ SERIAL, time_ TIME, ip VARCHAR(50) );
type = (id_type SERIAL, creation VARCHAR(50) , modification VARCHAR(50) , delete_ VARCHAR(50) , #id_audit_);
inscription = (id_inscription SERIAL, name VARCHAR(50) , surname VARCHAR(50) , date_ DATE);
grade = (id_grade SERIAL, grade VARCHAR(50) , coefficient NUMERIC(4,2)  , #id_subject);
school = (id_school SERIAL, name_of_the_school VARCHAR(50) , address VARCHAR(50) , telephone VARCHAR(50) , #id_inscription);
student = (id_student SERIAL, name VARCHAR(50) , surname VARCHAR(50) , date_of_birth DATE, #id_grade, #id_class);
personel = (id_personel SERIAL, name VARCHAR(50) , surname VARCHAR(50) , email VARCHAR(50) , phone_number VARCHAR(50) , #id_school, #id_user);
absence = (id_absence SERIAL, motif VARCHAR(50) , justification VARCHAR(50) , date_ DATE, #id_student);
authority_ = (#id_student, #id_parent);
teach = (#id_class, #id_subject);
present = (#id_parent, #id_subject, #id_teacher);
utilize = (#id_student, #id_user);
auditing = (#id_user, #id_audit_);
